CREATE OR REPLACE PROCEDURE PROC_CF008(P_NGAY_GG DATE,
                                       P_BRANCH  VARCHAR,
                                       P_CIF     VARCHAR,
                                       P_LOAI    VARCHAR,
                                       X_CURSOR  OUT SYS_REFCURSOR
                                       
                                       ) IS

   Procedure Name: PROC_CF008
   Report Code   : CF008
   Report Name   : Bao cao chi tiet thong tin khach hang
   Purpose       : Tong hop du lieu khach hang vao bang CF08_MAIN_DATAV2
   Business Ctx  : N/A
   Modified Date : 2026-05-22
   Developer     : ANHNTL
  */

  if P_LOAI != 'ALL' then
    INSERT INTO TMP.CF08_MAIN_DATAV2
      (branch_code,
       branch_name,
       CUSTOMER_NO,
       CUSTOMER_NAME,
       GIAY_TO,
       CMT,
       KENH,
       TONG_SD,
       NGAY_SINH,
       SDT_KH,
       E_MAIL,
       DIA_CHI,
       DUNG_TT,

       CHUA_CN,
       DA_CN,

       SAI_HO_TEN,
       SAI_NGAY_SINH,
       SD_K,
       SD_ST,
       CCCD_GANCHIP,
       SD_ST_GOIBCA,
       NGAYCAP,
       NOICAP,
       NGAY_HH
       
       )
      select A.local_branch,
             (select br_nm
                from SILVER.DIM_BRANCH
               where br_cd = A.local_branch
                 and rownum = 1) br_nm,
             a.customer_no,
             a.CUSTOMER_NAME1,
             a.GIAY_TO,
             a.UNIQUE_ID_VALUE CMND,
             a.Kenh,
             B.sodu_vnd tong_sd,
             to_char(a.NGAY_SINH, 'dd/MM/rrrr') NGAY_SINH,
             A.MOBILE_NUMBER,
             A.E_MAIL,
             a.DIA_CHI,
             (CASE
               WHEN c.field_val_29 = 'XT01' THEN
                'X'
               ELSE
                NULL
             END) sd7,
             (CASE
               WHEN c.field_val_29 = 'XT02' THEN
                'X'
               ELSE
                NULL
             END) sd8,
             (CASE
               WHEN c.field_val_29 = 'XT03' THEN
                'X'
               ELSE
                NULL
             END) sd9,
             (CASE
               WHEN c.field_val_29 = 'XT04' THEN
                'X'
               ELSE
                NULL
             END) sd10,

             null,

               WHEN T.CUS_NO IS NOT NULL THEN
                'X'
               ELSE
                NULL

               WHEN T.CO_STH = 'X' THEN
                'X'
               ELSE
                NULL
             END) SD_ST,

               WHEN T.CCCD_CHIP = 'X' THEN
                'X'
               ELSE
                NULL
             END) CCCD_GANCHI,
             (CASE
               WHEN T.CO_STH_GOIBCA = 'X' THEN
                'X'
               ELSE
                NULL
             END) SD_ST_GOIBCA,

             A.NOICAP,
             A.NGAYHH
        from ODS_RAW.ODS_BIP_CF08_CUS a,
             SILVER.MV_DIM_BRANCH mv,
             (select g.customer_no,
                     sum(nvl(sodu_vnd, 0)) sodu_vnd,
                     g.transaction_date
                From ODS_RAW.ods_tr_acctgl g
               where 1 = 1
                 and g.customer_type = 'I'
                 and g.sodu_vnd >= 0
                 and g.dr_gl like '42%'
                 AND G.transaction_date = P_NGAY_GG
               group by g.customer_no, g.transaction_date) b,
             (select c.field_val_29, REPLACE(c.rec_key, '~') CUSTOMER_NO
                from ODS_RAW.vndklb_CSZM_FUNCTION_USERDEF_FIELDS c
               where field_val_29 is not null) c,
             
             ODS_RAW.ODS_BIP_TABLE_DA06_BIO       G,
             ODS_RAW.ODS_BIP_TABLE_DA06_CCGANCHIP T
       where a.customer_no = b.customer_no(+)
         and a.customer_no = c.customer_no(+)
         and a.customer_no = T.CUS_NO(+)
         AND A.customer_no = G.CUS_NO(+)
         AND a.customer_no = decode(P_CIF, 'ALL', a.customer_no, P_CIF)
         AND c.field_val_29 = decode(P_LOAI, 'ALL', c.field_val_29, P_LOAI)
         and mv.BR_CD = a.local_branch
         AND mv.PARENT_CD = SUBSTR(P_BRANCH, 1, 3)
         and a.transaction_date = P_NGAY_GG;

      (branch_code,
       branch_name,
       CUSTOMER_NO,
       CUSTOMER_NAME,
       GIAY_TO,
       CMT,
       KENH,
       TONG_SD,
       NGAY_SINH,
       SDT_KH,
       E_MAIL,
       DIA_CHI,
       DUNG_TT,
       KHONG_TT,
       CHUA_CN,
       DA_CN,
       SAI_CCCD_CMND,
       SAI_HO_TEN,
       SAI_NGAY_SINH,
       SD_K,
       SD_ST,
       CCCD_GANCHIP,
       SD_ST_GOIBCA,
       NGAYCAP,
       NOICAP,
       NGAY_HH)

        from (select A.local_branch,
                     (select br_nm
                        from SILVER.DIM_BRANCH
                       where br_cd = A.local_branch
                         and rownum = 1) br_nm,
                     a.customer_no,
                     a.CUSTOMER_NAME1,
                     a.GIAY_TO,
                     a.UNIQUE_ID_VALUE CMND,
                     a.Kenh,
                     B.sodu_vnd tong_sd,
                     to_char(a.NGAY_SINH, 'dd/MM/rrrr') NGAY_SINH,
                     A.MOBILE_NUMBER,
                     A.E_MAIL,
                     a.DIA_CHI,
                     (CASE
                       WHEN c.field_val_29 = 'XT01' THEN
                        'X'
                       ELSE
                        NULL
                     END) sd7,
                     (CASE
                       WHEN c.field_val_29 = 'XT02' THEN
                        'X'
                       ELSE
                        NULL
                     END) sd8,
                     (CASE
                       WHEN c.field_val_29 = 'XT03' THEN
                        'X'
                       ELSE
                        NULL
                     END) sd9,
                     (CASE
                       WHEN c.field_val_29 = 'XT04' THEN
                        'X'
                       ELSE
                        NULL
                     END) sd10,
                     null SAI_CCCD_CMND,
                     null SAI_HO_TEN,
                     null SAI_NGAY_SINH,
                     (CASE
                       WHEN T.CUS_NO IS NOT NULL THEN
                        'X'
                       ELSE
                        NULL
                     END) SD_K,
                     (CASE
                       WHEN T.CO_STH = 'X' THEN
                        'X'
                       ELSE
                        NULL
                     END) SD_ST,
                     (CASE
                       WHEN T.CCCD_CHIP = 'X' THEN
                        'X'
                       ELSE
                        NULL
                     END) CCCD_GANCHI,
                     (CASE
                       WHEN T.CO_STH_GOIBCA = 'X' THEN
                        'X'
                       ELSE
                        NULL
                     END) SD_ST_GOIBCA,
                     A.NGAYCAP,
                     A.NOICAP,
                     A.NGAYHH
                from ODS_RAW.ODS_BIP_CF08_CUS a,
                     SILVER.MV_DIM_BRANCH mv,
                     (select g.customer_no,
                             sum(nvl(sodu_vnd, 0)) sodu_vnd,
                             g.transaction_date
                        From ODS_RAW.ods_tr_acctgl g
                       where 1 = 1
                         and g.customer_type = 'I'
                         and g.sodu_vnd >= 0
                         and g.dr_gl like '42%'
                         AND G.transaction_date = P_NGAY_GG
                       group by g.customer_no, g.transaction_date) b,
                     (select c.field_val_29,
                             REPLACE(c.rec_key, '~') CUSTOMER_NO
                        from ODS_RAW.vndklb_CSZM_FUNCTION_USERDEF_FIELDS c
                       where field_val_29 is not null) c,
                     ODS_RAW.ODS_BIP_TABLE_DA06_BIO G,
                     ODS_RAW.ODS_BIP_TABLE_DA06_CCGANCHIP T
               where a.customer_no = b.customer_no(+)
                 and a.customer_no = c.customer_no
                    
                 and a.customer_no = T.CUS_NO(+)
                 AND A.customer_no = G.CUS_NO(+)
                 AND a.customer_no =
                     decode(P_CIF, 'ALL', a.customer_no, P_CIF)
                 and mv.BR_CD = a.local_branch
                 AND mv.PARENT_CD = SUBSTR(P_BRANCH, 1, 3)
                 and a.transaction_date = P_NGAY_GG
              union all
              select A.local_branch,
                     (select br_nm
                        from SILVER.DIM_BRANCH
                       where br_cd = A.local_branch
                         and rownum = 1) br_nm,
                     a.customer_no,
                     a.CUSTOMER_NAME1,
                     a.GIAY_TO,
                     a.UNIQUE_ID_VALUE CMND,
                     a.Kenh,
                     B.sodu_vnd tong_sd,
                     to_char(a.NGAY_SINH, 'dd/MM/rrrr') NGAY_SINH,

                     A.E_MAIL,
                     a.DIA_CHI,

                     null sd8,
                     null sd9,
                     null sd10,
                     null,
                     null,
                     null,
                     (CASE
                       WHEN T.CUS_NO IS NOT NULL THEN
                        'X'
                       ELSE
                        NULL
                     END) SD_K,
                     (CASE
                       WHEN T.CO_STH = 'X' THEN
                        'X'
                       ELSE
                        NULL
                     END) SD_ST,
                     (CASE
                       WHEN T.CCCD_CHIP = 'X' THEN
                        'X'
                       ELSE
                        NULL
                     END) CCCD_GANCHI,
                     (CASE
                       WHEN T.CO_STH_GOIBCA = 'X' THEN
                        'X'
                       ELSE
                        NULL
                     END) SD_ST_GOIBCA,
                     A.NGAYCAP,
                     A.NOICAP,
                     A.NGAYHH
                from ODS_RAW.ODS_BIP_CF08_CUS a,
                     SILVER.MV_DIM_BRANCH mv,
                     (select g.customer_no,
                             sum(nvl(sodu_vnd, 0)) sodu_vnd,
                             g.transaction_date
                        From ODS_RAW.ods_tr_acctgl g
                       where 1 = 1
                         and g.customer_type = 'I'
                         and g.sodu_vnd >= 0
                         and g.dr_gl like '42%'
                         AND G.transaction_date = P_NGAY_GG
                       group by g.customer_no, g.transaction_date) b,
                     (select c.field_val_29,
                             REPLACE(c.rec_key, '~') CUSTOMER_NO
                        from ODS_RAW.vndklb_CSZM_FUNCTION_USERDEF_FIELDS c
                       where field_val_29 is null) c,
                     ODS_RAW.ODS_BIP_TABLE_DA06_BIO G,
                     ODS_RAW.ODS_BIP_TABLE_DA06_CCGANCHIP T
               where a.customer_no = b.customer_no(+)
                 and a.customer_no = c.customer_no
                 AND A.customer_no = G.CUS_NO(+)
                 and a.customer_no = T.CUS_NO(+)
                 AND a.customer_no =
                     decode(P_CIF, 'ALL', a.customer_no, P_CIF)
                 and mv.BR_CD = a.local_branch
                 AND mv.PARENT_CD = SUBSTR(P_BRANCH, 1, 3)
                 and a.transaction_date = P_NGAY_GG);
  end if;

END PROC_CF008;
